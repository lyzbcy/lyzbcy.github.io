# 静态网站"从夯到拉排名"功能实现文档

## 功能概述

"从夯到拉排名"是一个用于展示分级排名的交互式组件，支持：
- 5个等级的分层展示（夯、顶级、人上人、NPC、拉完了）
- 每个等级可包含多个项目卡片
- 卡片支持背景图片展示
- 点击卡片弹出模态框显示详细信息
- 自动生成详细说明区域
- 响应式设计，适配移动端和桌面端
- 精美的动画效果和玻璃态设计风格

#### 2.2 填充 Tier 列表

```javascript
function populateTierList() {
  const tierItems = {};
  
  // 获取所有 tier-row 容器
  document.querySelectorAll('.tier-row').forEach(row => {
    const label = row.querySelector('.tier-label');
    const items = row.querySelector('.tier-items');
    if (label && items) {
      tierItems[label.textContent.trim()] = items;
      items.innerHTML = '';  // 清空占位内容
    }
  });

  // 为每个项目创建卡片
  spots.forEach(spot => {
    const container = tierItems[spot.tier];
    if (!container) return;

    const card = document.createElement('div');
    card.className = 'spot-card';
    card.setAttribute('data-spot', spot.name);

    // 如果有背景图，添加样式类并设置 CSS 变量
    if (spot.bgImage) {
      card.classList.add('spot-card--with-bg');
      card.style.setProperty('--spot-bg-image', `url("${spot.bgImage}")`);
    }

    const label = document.createElement('span');
    label.className = 'spot-card__name';
    label.textContent = spot.name;
    card.appendChild(label);

    container.appendChild(card);
  });
}
```

```javascript
function showModal(spotName) {
  const data = spotData[spotName];
  if (!data || !modal || !modalTitle || !modalBody) {
    return;
  }

  modalTitle.textContent = data.name;
  modalBody.innerHTML = generateModalContent(data);

  // 动画显示
  modal.style.display = 'flex';
  modal.style.opacity = '0';
  const modalContent = modal.querySelector('.modal-content');
  if (modalContent) {
    modalContent.style.transform = 'translateY(30px) scale(0.95)';
    modalContent.style.opacity = '0';
    void modal.offsetWidth;  // 触发重排
    requestAnimationFrame(() => {
      modal.classList.add('show');
      modal.style.opacity = '1';
      modalContent.style.transform = 'translateY(0) scale(1)';
      modalContent.style.opacity = '1';
    });
  }

  document.body.style.overflow = 'hidden';
}

function hideModal() {
  if (!modal) return;
  const modalContent = modal.querySelector('.modal-content');
  modal.style.opacity = '0';
  if (modalContent) {
    modalContent.style.transform = 'translateY(30px) scale(0.95)';
    modalContent.style.opacity = '0';
  }
  setTimeout(() => {
    modal.classList.remove('show');
    modal.style.display = 'none';
    document.body.style.overflow = '';
  }, 300);
}
```

#### 2.6 初始化函数

```javascript
function initTravelTier() {
  const modal = document.getElementById('spotModal');
  const modalTitle = document.getElementById('modalTitle');
  const modalBody = document.getElementById('modalBody');
  const modalClose = document.getElementById('modalClose');
  const contentSection = document.querySelector('.content-section');

  // 填充列表
  populateTierList();

  // 绑定卡片点击事件
  const spotCards = document.querySelectorAll('.spot-card');
  spotCards.forEach(card => {
    card.addEventListener('click', function(e) {
      e.preventDefault();
      const spotName = this.getAttribute('data-spot');
      this.style.transform = 'scale(0.95)';
      setTimeout(() => {
        this.style.transform = '';
        showModal(spotName);
      }, 150);
    });

    // 触摸事件支持
    let touchStartTime = 0;
    card.addEventListener('touchstart', function() {
      touchStartTime = Date.now();
      this.style.transform = 'scale(0.98)';
    });

    card.addEventListener('touchend', function() {
      const touchDuration = Date.now() - touchStartTime;
      if (touchDuration < 300) {
        this.style.transform = 'scale(1.02)';
        setTimeout(() => {
          this.style.transform = '';
        }, 100);
      } else {
        this.style.transform = '';
      }
    });
  });

  // 关闭按钮
  if (modalClose) {
    modalClose.addEventListener('click', hideModal);
  }

  // 点击背景关闭
  if (modal) {
    modal.addEventListener('click', function(e) {
      if (e.target === modal) {
        hideModal();
      }
    });
  }

  // ESC 键关闭
  document.addEventListener('keydown', function(e) {
    if (e.key === 'Escape' && modal && modal.classList.contains('show')) {
      hideModal();
    }
  });

  // 生成详细说明
  if (contentSection) {
    contentSection.innerHTML = generateContentSectionHTML();
  }
}

// 页面加载完成后初始化
onReady(initTravelTier);
```

## 注意事项

1. **图片路径**：确保图片路径正确，使用 `encodeURIComponent()` 处理文件名中的特殊字符
2. **Tier 名称**：JavaScript 中的 `tier` 值必须与 HTML 中的 `.tier-label` 文本完全匹配
3. **CSS 变量**：背景图通过 CSS 变量 `--spot-bg-image` 设置，需要 `spot-card--with-bg` 类配合
4. **数据完整性**：每个项目必须包含 `name` 和 `tier` 字段，其他字段可选
5. **文件路径**：JavaScript 文件路径必须正确，使用 `defer` 属性确保 DOM 加载完成
6. **浏览器兼容性**：`backdrop-filter` 需要现代浏览器支持，旧版浏览器可能显示为普通半透明背景
