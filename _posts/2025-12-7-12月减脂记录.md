#### 计划说明

* **高碳日 (HC)：**
  * **匹配日：** 安排在训练强度最高、涉及最大肌群的日子（练背和练腿日）。
  * **目的：** 提供充足的糖原用于高强度训练，并确保训练后的超量恢复，最大程度支持增肌。

* **中碳日 (MC)：**
  * **匹配日：** 安排在训练强度中等的日子（练胸、手臂和肩部小肌群日）。
  * **目的：** 提供足够的能量支持训练，同时维持适度的热量赤字。

* **低碳日 (LC)：**
  * **匹配日：** 安排在休息日。
  * **目的：** 通过极低碳水摄入最大化脂肪氧化，提高胰岛素敏感性，加速燃脂。

* **蛋白质：**
  * 所有日期的蛋白质摄入量保持在高水平（150克），以确保在整个减脂期中肌肉得到充分保护和修复。

// 从碎碎念内容中提取心得总结
function extractSummaries() {
  const summaries = [];
  
  // 遍历所有笔记数据
  Object.keys(notesData).forEach(date => {
    const content = notesData[date];
    // 提取"💡 心得总结："部分
    const match = content.match(/💡\s*心得总结[：:]([\s\S]*?)(?=\n\n|$)/);
    if (match && match[1].trim()) {
      summaries.push({
        date: date,
        content: match[1].trim()
      });
    }
  });
  
  return summaries;
}

// 生成小贴纸HTML
function generateStickyNotes() {
  const container = document.getElementById('sticky-notes-container');
  const summaries = extractSummaries();
  
  if (summaries.length === 0) {
    container.innerHTML = '<div class="no-summaries">📭 还没有记录心得总结哦~<br>在每日碎碎念中添加"💡 心得总结："就会自动出现在这里！</div>';
    return;
  }
  
  container.innerHTML = summaries.map(summary => `
    <div class="sticky-note">
      <div class="sticky-date">📅 ${summary.date}</div>
      <div class="sticky-content">${summary.content}</div>
    </div>
  `).join('');
}

    modal.classList.add('active');
    document.body.style.overflow = 'hidden';
  }
}

// 关闭心得总结板
function closeSummaryBoard() {
  const modal = document.getElementById('summary-modal');
  if (modal) {
    modal.classList.remove('active');
    document.body.style.overflow = 'auto';
  }
}

// 点击背景关闭
function closeSummaryOnBackdrop(event) {
  if (event.target.classList.contains('summary-modal')) {

  }
}

    document.body.appendChild(lightbox);
    
    // 点击背景关闭(只添加一次)
    lightbox.addEventListener('click', function(e) {
      if (e.target === lightbox) {
        closeLightbox();
      }
    });
  }
  
  // 设置图片和标题
  const img = lightbox.querySelector('.lightbox-image');
  const title = lightbox.querySelector('.lightbox-title');
  img.src = imageSrc;
  img.alt = imageTitle || '';
  title.textContent = imageTitle || '';
  
  // 显示lightbox
  lightbox.classList.add('active');
  document.body.style.overflow = 'hidden';
}

function closeLightbox() {
  const lightbox = document.getElementById('image-lightbox');
  if (lightbox) {
    lightbox.classList.remove('active');
    document.body.style.overflow = 'auto';
  }
}
